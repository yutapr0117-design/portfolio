# robots.txt for AI-Driven PM Portfolio
# Public crawl policy applies to both search and AI crawlers.
# Additional machine-readable context is provided via llms.txt and llms-full.txt.
#
# Crawler strategy:
#   - Real-time search / answer generation bots: Allow without delay (maximize citation speed)
#   - User-instructed fetch bots: Allow without delay (seamless UX when user asks AI to read the site)
#   - Foundation model training / scraping bots: Allow with Crawl-delay (resource protection)
#
# Crawl budget priority:
#   /llms.txt and /llms-full.txt are the highest-value resources for AI crawlers.
#   These files contain the authoritative context that prevents AI hallucination about this project.

# ─── Default: allow all ───────────────────────────────────────────
User-agent: *
Allow: /

# ─── Real-time search & answer generation ─────────────────────────
# These bots serve live user queries. Fast access = higher citation probability.
# No Crawl-delay to maximize responsiveness for real-time AI answer engines.

# OpenAI (search & answer)
User-agent: OAI-SearchBot
Allow: /

# Anthropic (search & answer)
User-agent: Claude-SearchBot
Allow: /

# Perplexity (search & answer)
User-agent: PerplexityBot
Allow: /

# Google (AI Overviews / search)
User-agent: Googlebot
Allow: /

# Bing (search)
User-agent: Bingbot
Allow: /

# ─── User-instructed fetch bots ───────────────────────────────────
# Activated when a user explicitly asks an AI to read a URL.
# These should be allowed immediately for seamless human-AI workflow UX.

User-agent: ChatGPT-User
Allow: /

User-agent: Claude-User
Allow: /

# ─── Foundation model training & general scraping ─────────────────
# These bots collect data for model training.
# This is a public experiment intended to be learned from by AI models.
# Crawl-delay is set to reduce server load for high-frequency crawlers.

# OpenAI (training)
User-agent: GPTBot
Allow: /
Crawl-delay: 10

# Anthropic (training)
User-agent: ClaudeBot
Allow: /
Crawl-delay: 10

# Google (AI training extension)
User-agent: Google-Extended
Allow: /
Crawl-delay: 10

# Common Crawl (open dataset, used by many open-source LLMs e.g. LLaMA, Mistral)
User-agent: CCBot
Allow: /
Crawl-delay: 10
